for i in ID:
    data = result[(result.Stkcd == i)]
    idvol=[]
    Stkcd=[]
    date=[]
    for j in range(len(list(set(data['yearMonth'])))-11):
        kaishi=data.iloc[0,5]+relativedelta(months=(j))
        end=data.iloc[0,5]+relativedelta(months=(11+j))
        data1=data[kaishi:end]
        if len(data1['Trddt'])>=120:
            from sklearn.linear_model import LinearRegression 
            lm=LinearRegression() 
            data1 = data.fillna(0) # 0 
            X = data1['Cdretmdeq']
            Y = data1['Dretnd']
            Y=Y.values.reshape(-1, 1)
            X=X.values.reshape(-1, 1)
            lm.fit(X,Y) #Fitting 
            Yhat=lm.predict(X) #Predicting 
            data1['resid']=Y-Yhat
            a=data1['resid'].std(ddof=1)
            idvol.append(a) 
        else:
            a=0
            idvol.append(a)
        date0=data1[-1:]
        date1=list(date0['yearMonth'])
        date.append(date1)
        stkcd=i
        Stkcd.append(stkcd)
    A = pd.DataFrame(Stkcd, columns=['Stkcd'])
    B = pd.DataFrame(date, columns=['date'])
    C = pd.DataFrame(idvol, columns=['idvol'])
    m= pd.concat([A, B, C], axis=1)
    idvol_01=idvol_01.append(m)

def _ttest_finish(df, t):
    #from scipy.stats
    '''
    :param df:
    :param t: t
    '''
    prob = distributions.t.sf(np.abs(t), df) * 2  # use np.abs to get upper tail
    if t.ndim == 0:
        t = t[()]
    return t, prob

NWt_1sampleResult = namedtuple('NWT_1sampResult', ('statistic', 'pvalue'))
def nwttest_1samp(a, popmean, axis=0,L=1):
    '''
    
    :param a: 
    :param popmean: u0
    :param axis: 
    :param L: lag 1
    :return: nw-tp
    '''
    a = np.array(a)
    N = len(a)
    df = N-1
    e = a - np.mean(a)
    residuals = np.sum(e**2)
    Q = 0
    for i in range(L):
        w_l = 1 - (i+1)/(1+L)
        for j in range(1,N):
            Q += w_l*e[j]*e[j-(i+1)]
    S = residuals + 2*Q
    nw_var = S/N
    d = np.mean(a,axis) - popmean
    nw_sd = np.sqrt(nw_var / float(df))
    with np.errstate(divide='ignore', invalid='ignore'):
        t = np.divide(d, nw_sd)
    t,prob = _ttest_finish(df,t)

    return NWt_1sampleResult(t,prob)
